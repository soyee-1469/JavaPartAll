*집합연산자
-합집합 연산자
UNION : 연결된 SELECT문의 결과 값을 합집합으로 묶어준다. 
-결과값의 중복은 제거된다. 
-- 열의수가 같고 값이 다르면
SELECT EMPNO, ENAME, SAL, DEPTNO FROM EMP
WHERE DEPTNO = 10
UNION
SELECT EMPNO, ENAME, SAL, DEPTNO FROM EMP
WHERE DEPTNO = 20;

-- 열의수가 다를떄
SELECT EMPNO, ENAME, SAL, DEPTNO FROM EMP
WHERE DEPTNO = 10
UNION
SELECT EMPNO, SAL,ENAME , DEPTNO FROM EMP
WHERE DEPTNO = 10;
-- 열이 일치하지는 않지만 타입이 같은 경우 출력은 가능하나 신뢰도가 낮음

UNION ALL 연결된 SELECT문의 결과 값을 합집합으로 묶어준다. 
0 중복된 결과 값도 제거없이 모두 출력
SELECT EMPNO, ENAME, SAL, DEPTNO FROM EMP
WHERE DEPTNO = 10
UNION  ALL
SELECT EMPNO, ENAME, SAL, DEPTNO FROM EMP
WHERE DEPTNO = 20;
